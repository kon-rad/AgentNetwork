    **Create agent-server/src/channels/webapp/index.ts:**

    This is the core NC-02 deliverable. Implement the full Channel interface with Express HTTP server:

    ```typescript
    // src/channels/webapp/index.ts
    import express from 'express'
    import * as fs from 'fs'
    import * as path from 'path'
    import chokidar from 'chokidar'
    import { registerChannel } from '../registry'
    import type { Channel, ChannelOpts, IncomingMessage } from '../../types'
    import { WEBAPP_PORT, WEBAPP_SHARED_SECRET } from '../../config'

    // Map of agentJid -> active SSE response object (one per open stream)
    const sseClients = new Map<string, express.Response>()

    registerChannel('webapp', (opts: ChannelOpts): Channel | null => {
      if (!process.env.WEBAPP_PORT) {
        console.warn('[webapp] WEBAPP_PORT not set — webapp channel disabled')
        return null
      }

      const app = express()
      app.use(express.json())

      // Shared secret auth middleware (NC-06)
      app.use((req, res, next) => {
        const secret = req.headers['x-shared-secret']
        if (!WEBAPP_SHARED_SECRET || secret !== WEBAPP_SHARED_SECRET) {
          return res.status(401).json({ error: 'unauthorized' })
        }
        next()
      })

      // POST /message — receive message from Next.js (NC-02)
      app.post('/message', async (req, res) => {
        const { agentId, message, sessionToken } = req.body
        if (!agentId || !message) {
          return res.status(400).json({ error: 'agentId and message required' })
        }
        const jid = `${agentId}@webapp`
        const incoming: IncomingMessage = {
          jid,
          text: message,
          sender: sessionToken || 'anonymous',
          senderName: 'user',
          timestamp: Date.now(),
        }
        try {
          await opts.onMessage(incoming)
          res.json({ queued: true, jid })
        } catch (err) {
          console.error('[webapp] onMessage error:', err)
          res.status(500).json({ error: 'failed to queue message' })
        }
      })

      // GET /stream/:agentId — SSE stream for agent response (NC-02)
      app.get('/stream/:agentId', (req, res) => {
        const { agentId } = req.params
        const jid = `${agentId}@webapp`

        res.setHeader('Content-Type', 'text/event-stream')
        res.setHeader('Cache-Control', 'no-cache')
        res.setHeader('Connection', 'keep-alive')
        res.setHeader('X-Accel-Buffering', 'no')  // disable Nginx/Caddy buffering
        res.flushHeaders()

        sseClients.set(jid, res)

        // Send a heartbeat every 15s to keep connection alive through proxies
        const heartbeat = setInterval(() => {
          res.write(': heartbeat\n\n')
        }, 15000)

        req.on('close', () => {
          clearInterval(heartbeat)
          sseClients.delete(jid)
        })
      })

      // POST /register-group — register a new agent group (NC-07)
      app.post('/register-group', async (req, res) => {
        const { agentId, folder, claudeMdContent } = req.body
        if (!agentId || !folder) {
          return res.status(400).json({ error: 'agentId and folder required' })
        }
        try {
          // Insert into NanoClaw's SQLite registered_groups table via opts.db
          opts.db.prepare(
            `INSERT OR IGNORE INTO registered_groups (jid, folder, created_at)
             VALUES (?, ?, datetime('now'))`
          ).run(`${agentId}@webapp`, folder)

          // Create workspace directory with CLAUDE.md
          const groupDir = path.join(process.cwd(), 'data', 'groups', folder)
          fs.mkdirSync(groupDir, { recursive: true })

          if (claudeMdContent) {
            fs.writeFileSync(path.join(groupDir, 'CLAUDE.md'), claudeMdContent, 'utf8')
          }

          res.json({ success: true, groupId: agentId, folder })
        } catch (err) {
          console.error('[webapp] register-group error:', err)
          res.status(500).json({ error: 'failed to register group' })
        }
      })

      app.listen(WEBAPP_PORT, () => {
        console.log(`[webapp] channel listening on port ${WEBAPP_PORT}`)
      })

      return {
        name: 'webapp',
        connect: async () => {
          console.log('[webapp] channel connected')
        },
        sendMessage: async (jid: string, text: string) => {
          // Called by NanoClaw after agent turn completes (via IPC watcher in router.ts)
          const client = sseClients.get(jid)
          if (client) {
            client.write(`data: ${JSON.stringify({ text, done: false })}\n\n`)
          } else {
            console.warn(`[webapp] no SSE client for jid ${jid}`)
          }
        },
        sendDone: async (jid: string) => {
          // Signal stream completion
          const client = sseClients.get(jid)
          if (client) {
            client.write(`data: ${JSON.stringify({ done: true })}\n\n`)
            client.end()
            sseClients.delete(jid)
          }
        },
        isConnected: () => true,
        ownsJid: (jid: string) => jid.endsWith('@webapp'),
        disconnect: async () => {
          // Close all open SSE connections
          for (const [jid, res] of sseClients.entries()) {
            res.write(`data: ${JSON.stringify({ error: 'server shutting down' })}\n\n`)
            res.end()
          }
          sseClients.clear()
        },
      }
    })
    ```

    Note: If the `Channel` interface in upstream types.ts does not include `sendDone` or `db`, adapt accordingly — check the actual types.ts after copying from upstream and adjust the implementation to match the real interface.

    ```typescript
    // src/supabase-logger.ts
    // Writes agent turn events to Supabase agent_events table.
    // Guarded: if SUPABASE_URL or key are missing, logs a warning and returns a no-op.

    import { createClient, SupabaseClient } from '@supabase/supabase-js'

    let supabase: SupabaseClient | null = null

    if (process.env.SUPABASE_URL && process.env.SUPABASE_SERVICE_ROLE_KEY) {
      supabase = createClient(
        process.env.SUPABASE_URL,
        process.env.SUPABASE_SERVICE_ROLE_KEY
      )
      console.log('[supabase-logger] initialized')
    } else {
      console.warn('[supabase-logger] SUPABASE_URL or SUPABASE_SERVICE_ROLE_KEY not set — event logging disabled')
    }

    export interface AgentEvent {
      agent_id: string
      event_type: 'turn_start' | 'turn_complete' | 'tool_call' | 'llm_call' | 'error'
      payload: Record<string, unknown>
      created_at?: string
    }

    export async function logEvent(event: AgentEvent): Promise<void> {
      if (!supabase) return
      const { error } = await supabase
        .from('agent_events')
        .insert({
          ...event,
          created_at: event.created_at || new Date().toISOString(),
        })
      if (error) {
        // Log but don't throw — observability failure must not crash agent turns
        console.error('[supabase-logger] insert error:', error.message)
      }
    }
    ```
