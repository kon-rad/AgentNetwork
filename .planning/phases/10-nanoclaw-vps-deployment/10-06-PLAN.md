  <action>
    Create `agent-server/scripts/e2e-test.sh`. This script tests the full NanoClaw webapp channel from the outside.

    Read TRANSPORT.md to get:
    - NANOCLAW_URL (e.g., https://nanoclaw.example.com OR http://10.0.0.1:3000)
    - Transport type (HTTPS or WireGuard)

    Read VPS .env or 10-04-SUMMARY.md to get the WEBAPP_SHARED_SECRET.

    ```bash
    #!/bin/bash
    set -euo pipefail

    # Configuration — read from environment or set defaults for local testing
    NANOCLAW_URL="${NANOCLAW_URL:-https://nanoclaw.YOURDOMAIN.com}"
    SHARED_SECRET="${WEBAPP_SHARED_SECRET:-}"
    TEST_AGENT_ID="test-agent-$(date +%s)"
    TEST_FOLDER="test-group-$(date +%s)"

    if [ -z "$SHARED_SECRET" ]; then
      echo "ERROR: WEBAPP_SHARED_SECRET env var required"
      exit 1
    fi

    echo "=== NanoClaw E2E Test ==="
    echo "URL: $NANOCLAW_URL"
    echo ""

    # Test 1: Auth rejection (NC-06)
    echo "[1/5] Testing auth rejection (no secret)..."
    RESPONSE=$(curl -s -o /dev/null -w "%{http_code}" -X POST "$NANOCLAW_URL/message" \
      -H "Content-Type: application/json" \
      -d '{"agentId":"test","message":"hello"}')
    [ "$RESPONSE" = "401" ] && echo "PASS: 401 returned without secret" || echo "FAIL: Expected 401, got $RESPONSE"

    # Test 2: POST /register-group (NC-07)
    echo ""
    echo "[2/5] Testing POST /register-group..."
    RESPONSE=$(curl -s -X POST "$NANOCLAW_URL/register-group" \
      -H "Content-Type: application/json" \
      -H "x-shared-secret: $SHARED_SECRET" \
      -d "{\"agentId\":\"$TEST_AGENT_ID\",\"folder\":\"$TEST_FOLDER\",\"claudeMdContent\":\"# Test Agent\\n\\nYou are a test agent. Respond with exactly: NANOCLAW_E2E_OK\"}")
    echo "register-group response: $RESPONSE"
    echo "$RESPONSE" | grep -q "success.*true" && echo "PASS: group registered" || echo "FAIL: unexpected response"

    # Test 3: POST /message (NC-02)
    echo ""
    echo "[3/5] Testing POST /message..."
    RESPONSE=$(curl -s -X POST "$NANOCLAW_URL/message" \
      -H "Content-Type: application/json" \
      -H "x-shared-secret: $SHARED_SECRET" \
      -d "{\"agentId\":\"$TEST_AGENT_ID\",\"message\":\"Say exactly: NANOCLAW_E2E_OK\",\"sessionToken\":\"e2e-test\"}")
    echo "message response: $RESPONSE"
    echo "$RESPONSE" | grep -q "queued.*true" && echo "PASS: message queued" || echo "FAIL: unexpected response"

    # Test 4: GET /stream/:agentId — wait for SSE response (NC-02)
    echo ""
    echo "[4/5] Testing GET /stream/$TEST_AGENT_ID (SSE — waiting up to 60s for agent turn)..."
    echo "  Watching Docker containers for agent spawn..."

    # Open SSE stream in background, capture output for 60s
    SSE_OUTPUT=$(curl -sN --max-time 60 \
      -H "x-shared-secret: $SHARED_SECRET" \
      "$NANOCLAW_URL/stream/$TEST_AGENT_ID" 2>&1 &)
    SSE_PID=$!

    # Poll docker ps on VPS to verify container spawn (requires SSH access)
    echo "  Waiting 5s for container spawn..."
    sleep 5
    # Note: docker ps check is manual — output will show in VPS docker logs

    # Wait for SSE result
    wait $SSE_PID 2>/dev/null || true
    echo "SSE output: $SSE_OUTPUT"

    if echo "$SSE_OUTPUT" | grep -q "NANOCLAW_E2E_OK\|data:"; then
      echo "PASS: SSE received agent response"
    else
      echo "INFO: SSE may have timed out — check VPS docker logs for container activity"
      echo "  Run: ssh deploy@VPS_IP 'docker ps -a | grep nanoclaw-agent'"
    fi

    echo ""
    echo "=== Test complete ==="
    echo "If SSE test timed out: the webapp channel POST/auth/register are confirmed working."
    echo "Full round-trip (container spawn + SSE response) may need VPS docker log inspection."
    ```

    Make the script executable: `chmod +x agent-server/scripts/e2e-test.sh`
